/*
  ERP_TEST v1.0.7 — Sys_Roles 增加 Permissions（菜单 path JSON 数组）
  - 列类型：NVARCHAR(MAX)，存 JSON，如 ["system/operator","canteen"] 或 ["*"] 表示全部
  - NULL：应用层视为「未限制」，兼容旧数据（登录后仍能看到全部菜单）
*/

IF COL_LENGTH('dbo.Sys_Roles', 'Permissions') IS NULL
BEGIN
  ALTER TABLE dbo.Sys_Roles ADD Permissions NVARCHAR(MAX) NULL;
END;

PRINT N'v1.0.7：Sys_Roles.Permissions 列已就绪（可为 NULL）。';
